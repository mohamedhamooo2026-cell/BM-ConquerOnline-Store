info.category : 'other'; groups[cat].elements.push({ key, name: info.name || key }); } order.forEach(catKey => { const group = groups[catKey]; if (group && group.elements.length > 0) { const optgroup = document.createElement("optgroup"); optgroup.label = group.label; group.elements.forEach(el => { const opt = document.createElement("option"); opt.value = el.key; opt.textContent = el.name; optgroup.appendChild(opt); }); select.appendChild(optgroup); } }); } function calculatePrice() { const service = document.getElementById("serviceSelector").value; const server = document.getElementById("serverSelector").value; const op = document.getElementById("operationSelector").value; const amount = document.getElementById("amountInput").value; if(globalData && globalData[service] && globalData[service][server]) { const price = globalData[service][server][op] || 0; // ملاحظة: الـ CPs مخزن كـ "سعر الحبة"، لذا الحساب تلقائي وصحيح document.getElementById("cashInput").value = (amount * price).toFixed(2); } } // ربط الأحداث للتحديث التلقائي document.getElementById("serviceSelector").onchange = updateDropdown; document.getElementById("serverSelector").onchange = calculatePrice; document.getElementById("operationSelector").onchange = calculatePrice; document.getElementById("amountInput").oninput = calculatePrice;
